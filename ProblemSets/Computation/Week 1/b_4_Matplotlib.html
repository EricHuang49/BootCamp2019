# Problem 1
def var_random_normal(n):
    x = np.random.normal(size=(n,n))
    y = np.mean(x,axis = 1)
    return np.var(y)

x = [var_random_normal(i) for i in np.arange(100,1100,100)]
plt.plot(np.arange(100,1100,100),x)
plt.xlabel('n')
plt.ylabel('var of mean of n random normal variables')
plt.show()

def plot_fun(a,b):
    x = np.linspace(a,b,100)
    plt.plot(x,np.sin(x),label='sin')
    plt.plot(x,np.cos(x),label='cos')
    plt.plot(x,np.arctan(x),label='arctan')
    plt.legend()
    
plot_fun(-2*np.pi,2*np.pi)
